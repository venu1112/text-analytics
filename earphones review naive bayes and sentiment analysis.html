corpus=[]
for i in range(0,14337):
    ReviewBody=re.sub('[^a-zA-Z]',' ',earphones['ReviewBody'][i])
    ReviewBody=ReviewBody.lower()
    ReviewBody=ReviewBody.split()
    ps=PorterStemmer()
    ReviewBody=[ps.stem(word) for word in ReviewBody if not word in set(stopwords.words('english'))]
    ReviewBody=' '.join(ReviewBody)
    corpus.append(ReviewBody)
